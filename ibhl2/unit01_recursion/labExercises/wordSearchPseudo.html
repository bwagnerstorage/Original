   method boolean findWord(String WORD, int R, int C, int INDEX)
      if NOT inBounds(R,C) then
         return false
      end if
      if INDEX >= WORD.length() then  
         return true
      end if
      if PUZZLE[R][C] = WORD.substring(INDEX, INDEX + 1) then
         if findWord(WORD, R+1, C, INDEX+1)    or
