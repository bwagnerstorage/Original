Add the following code to the SmallAsteroid class.

import greenfoot.*;

// Makes a small asteroid with dimensions 40 x 40
public class SmallAsteroid extends Asteroid
{
    public SmallAsteroid()
    {
        GreenfootImage image = new GreenfootImage(40,40);  // creates a blank image size 40 X 40
        image.setColor(Color.WHITE);
        
        image.drawLine(1,20,20,1); 
        image.drawLine(20,1,35,15);
        image.drawLine(35,15,34,39);  
        image.drawLine(34,39,15,35);  
        image.drawLine(15,35,18,25);  
        image.drawLine(18,25,1,20);  
           
        setImage(image);           // set image
    }    
}
