Below are recursive algorithms for both the linear search and binary search.

Linear Search

LINEAR_SEARCH(LIST, LENGTH, KEY)
   if LENGTH < 0 then
      return false
   end if
   if LIST[LENGTH] = KEY then
      return true
   else
      return LINEAR_SEARCH(LIST, LENGTH - 1, KEY)
   end if
end method

Binary Search

LEFT = 0
RIGHT = LIST.LENGTH - 1
BINARY_SEARCH(LIST, LEFT, RIGHT, KEY)
   MIDPOINT = (LEFT+RIGHT) / 2
   if LEFT > RIGHT then
      return false
   end if
   if LIST[MIDPOINT] = KEY then
      return true
   end if
   if LIST[MIDPOINT] < KEY then
      return BINARY_SEARCH(LIST, MIDPOINT + 1, RIGHT, KEY)
   else
      return BINARY_SEARCH(LIST, LEFT, MIDPOINT - 1, KEY)
   end if
end method
